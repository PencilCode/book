# Random Tree example by mckoss
bk 260

Math.seedrandom 8

rtree = (size) ->
  if size < 5
    return
  fd size
  rt 30
  rtree(size * (random(5) + 5) / 10)
  lt 60
  rtree(size * (random(5) + 5) / 10)
  rt 30
  bk size
